/* Clear and prepare database */
drop table poi;
drop table poi_tag;

.separator "\t"

create table poi(
id INTEGER NOT NULL PRIMARY KEY,
uid INTEGER NOT NULL,
minX FLOAT NOT NULL,
maxX FLOAT NOT NULL,
minY FLOAT NOT NULL,
maxY FLOAT NOT NULL);
.import poi_translated.tsv poi

create table poi_tag(
id INTEGER NOT NULL REFERENCES poi,
key TEXT NOT NULL,
value TEXT NOT NULL);
.import poi_tag.tsv poi_tag
